\footnotesize{\textbf{Algorithm - }Insertion($A$)}\\
\item[]
\footnotesize{\textbf{Input:} an n-element un-sorted array A of integer values.}
\item[]
\footnotesize{\textbf{Output:} an n-element sorted array A of integer values.} 

\item[]
\scalebox{0.9}{
\begin{algorithm}[H]
\begin{algorithmic}[1]
\FOR{$i=1$ to $n$}
\STATE $key \gets A[i]$
\STATE $j \gets i-1$
\WHILE{$j >= 0$ and $A[j] > key$)}
\STATE $A[j+1] \gets A[j]$
\STATE $j \gets j-1$
\ENDWHILE
\STATE $A[j+1] \gets key$
\ENDFOR
\end{algorithmic}
\label{alg:seq}
\end{algorithm}
}
\end{itemize}
